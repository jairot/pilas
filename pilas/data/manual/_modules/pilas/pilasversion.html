[docs]def compare(v0, v1):
    """Compara dos versiones de pilas.

    Sus posibles retornos son

    - **-1** si ``v0`` < ``v1``.
    - **0** si ``v0`` == ``v1``.
    - **1** si ``v0`` > ``v1``.

    :param v0: primer versión a comparar.
    :type v0: str
    :param v1: segunda versión a comparar.
    :type v1: str

    """
    v0 = v0.split(".")
    v1 = v1.split(".")
    return -1 if v0 < v1 else 0 if v0 == 1 else 1
